# Provision cloud
juju add-cloud --client cathouse
juju bootstrap --bootstrap-series=jammy --constraints tags=juju cathouse cathouse-controller
juju add-model --config default-series=jammy openstack

# Add machine
juju add-machine ssh:jacques@192.168.64.3

# Switch context
juju switch cathouse-controller:openstack

# Install Ceph
juju deploy -n 1 --to 1 --channel reef/stable --config ceph-osd.yaml --constraints tags=compute ceph-osd

# Install Nova
juju deploy -n 1 --to 1 --channel 2023.2/stable --config nova-compute.yaml nova-compute

# Install MySQL cluster
juju deploy -n 3 --to lxd:1, lxd:1, lxd:1 --channel 8.0/stable mysql-innodb-cluster

# Install Vault
juju deploy --to lxd:1 --channel 1.8/stable vault

# Join Vault to MySQL cluster
juju deploy --channel 8.0/stable mysql-router vault-mysql-router
juju integrate vault-mysql-router:db-router mysql-innodb-cluster:db-router
juju integrate vault-mysql-router:shared-db vault:shared-db

# Initialize vault
